\column{0.50\textwidth}

\begin{frame}{Invoice QR cho hóa đơn}
\begin{columns}[T,onlytextwidth]
\column{0.52\textwidth}
\begin{enumerate}
  \item Cư dân chọn một hoặc nhiều bill cùng căn hộ.
  \item Backend kiểm tra bill còn \tech{UNPAID}/\tech{OVERDUE} và chưa bị khóa bởi invoice khác.
  \item Hệ thống sinh \tech{INVyyyyMMddNNN}, mã tham chiếu \tech{PAY...}, QR URL.
  \item Bill được khóa bằng \tech{invoice\_id}; snapshot giữ dữ liệu tại thời điểm tạo invoice.
\end{enumerate}

\begin{frame}{Webhook SePay}
\begin{columns}[T,onlytextwidth]
\column{0.50\textwidth}
\begin{itemize}
  \item Chỉ xử lý giao dịch vào: \tech{transferType = in}.
  \item Chống trùng webhook bằng \tech{transactionId}.
  \item Tách mã từ nội dung chuyển khoản: \tech{PAY... INV...}.
  \item Kiểm tra invoice tồn tại, đúng mã, đúng trạng thái, đúng số tiền.
\end{itemize}
\column{0.45\textwidth}
\begin{tabularx}{\linewidth}{>{\bfseries}l X}
\toprule
Kết quả & Xử lý\\
\midrule
SUCCESS & Tạo payment, đánh dấu invoice/bill đã thanh toán, gửi thông báo\\
FAILED & Lưu lý do: sai mã, sai tiền, hết hạn, đã thanh toán\\
Ignored & Bỏ qua giao dịch không liên quan hoặc chiều tiền không phù hợp\\
\bottomrule
\end{tabularx}
\end{columns}
\end{frame}

\begin{frame}{Thanh toán thủ công và hết hạn invoice}
\begin{columns}[T,onlytextwidth]
\column{0.47\textwidth}
\textbf{\textcolor{hustprimary}{Thanh toán thủ công}}
\begin{itemize}
  \item Admin có thể ghi nhận payment khi đối soát ngoài hệ thống.
  \item Luồng vẫn đi qua service để cập nhật invoice, bill và thông báo.
  \item Payment lưu phương thức, số tiền, người ghi nhận và thời điểm.
\end{itemize}
\column{0.48\textwidth}
\textbf{\textcolor{hustprimary}{Hết hạn invoice}}
\begin{itemize}
  \item Job chạy định kỳ để phát hiện invoice \tech{PENDING} quá hạn.
  \item Invoice chuyển sang \tech{EXPIRED}.
  \item Bill bị khóa được giải phóng để cư dân tạo invoice mới.
\end{itemize}
\end{columns}
\end{frame}
