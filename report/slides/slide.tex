\section{Luồng nghiệp vụ chính}

\begin{frame}{Luồng xác thực}
\begin{columns}[T,onlytextwidth]
\column{0.50\textwidth}
\begin{enumerate}
  \item Người dùng đăng nhập bằng email/mật khẩu hoặc Google OAuth.
  \item Server trả JWT nếu thông tin hợp lệ.
  \item Giao diện lưu token và thêm header \tech{Authorization}.
  \item Khi token lỗi, interceptor xóa phiên và đưa về màn hình đăng nhập.
\end{enumerate}

\begin{frame}{Quản lý tài khoản và căn hộ}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{itemize}
  \item Quản trị viên tạo, cập nhật, xóa hoặc đặt lại mật khẩu tài khoản.
  \item Căn hộ lưu thông tin cơ bản và liên kết các cư dân thuộc căn hộ.
  \item Cư dân có trang ``my apartment'' để xem thông tin căn hộ và thành viên.
\end{itemize}

\begin{frame}{Tạo hóa đơn và theo dõi công nợ}
\begin{columns}[T,onlytextwidth]
\column{0.52\textwidth}
\begin{enumerate}
  \item Quản trị viên cấu hình \tech{bill\_templates}: loại phí, mô tả, số tiền, chu kỳ.
  \item Có thể tạo hóa đơn lẻ hoặc sinh hàng loạt theo danh sách căn hộ.
  \item Cư dân xem hóa đơn của căn hộ mình cùng trạng thái hiện tại.
  \item Tác vụ định kỳ có thể chuyển hóa đơn quá hạn sang \tech{OVERDUE}.

\begin{frame}{Tạo mã QR thanh toán hóa đơn}
\begin{columns}[T,onlytextwidth]
\column{0.52\textwidth}
\begin{enumerate}
  \item Cư dân chọn một hoặc nhiều hóa đơn cùng căn hộ.
  \item Server kiểm tra hóa đơn còn \tech{UNPAID}/\tech{OVERDUE} và chưa bị khóa bởi invoice khác.
  \item Hệ thống sinh \tech{INVyyyyMMddNNN}, mã tham chiếu \tech{PAY...}, QR URL.
  \item Hóa đơn được khóa bằng \tech{invoice\_id}; snapshot giữ dữ liệu tại thời điểm tạo invoice.
\end{enumerate}

\begin{frame}{Webhook SePay}
\begin{columns}[T,onlytextwidth]
\column{0.50\textwidth}
\begin{itemize}
  \item Chỉ xử lý giao dịch vào: \tech{transferType = in}.
  \item Chống xử lý trùng bằng \tech{transactionId}.
  \item Tách mã từ nội dung chuyển khoản: \tech{PAY... INV...}.
  \item Kiểm tra invoice tồn tại, đúng mã, đúng trạng thái, đúng số tiền.
\end{itemize}
\column{0.45\textwidth}
\footnotesize
\begin{tabularx}{\linewidth}{>{\bfseries\RaggedRight\arraybackslash}P{1.55cm} Q}
\toprule
Kết quả & Xử lý\\
\midrule
SUCCESS & Ghi nhận thanh toán, cập nhật invoice/hóa đơn, gửi thông báo\\
FAILED & Lưu lý do: sai mã, sai tiền, hết hạn, đã thanh toán\\
IGNORED & Bỏ qua giao dịch không liên quan hoặc không phải tiền vào\\
\bottomrule
\end{tabularx}
\end{columns}
\end{frame}

\begin{frame}{Thanh toán thủ công và invoice quá hạn}
\begin{columns}[T,onlytextwidth]
\column{0.47\textwidth}
\textbf{\textcolor{hustprimary}{Thanh toán thủ công}}
\begin{itemize}
  \item Quản trị viên có thể ghi nhận thanh toán sau khi đối soát ngoài hệ thống.
  \item Thao tác vẫn đi qua service để cập nhật invoice, hóa đơn và thông báo.
  \item Bản ghi thanh toán lưu phương thức, số tiền, người ghi nhận và thời điểm.
\end{itemize}
\column{0.48\textwidth}
\textbf{\textcolor{hustprimary}{Invoice quá hạn}}
\begin{itemize}
  \item Tác vụ định kỳ phát hiện invoice \tech{PENDING} quá hạn.
  \item Invoice chuyển sang \tech{EXPIRED}.
  \item Hóa đơn bị khóa được giải phóng để cư dân tạo invoice mới.
\end{itemize}
\end{columns}
